Now we are see about how to use the chart tool in vb.net
For that we need visual studio 2010 or 2011
1)create a empty form
2)Add chart control to the form




3)set the series name
4)Then create one data base i choose ms access


4) In page load event type the following code

Imports System.Data.OleDb
Public Class Form1
    Dim con As New OleDbConnection("Provider=Microsoft.ACE.OLEDB.12.0;Data Source=C:\Users\ravin\Desktop\college\sub.accdb")
    Dim cmd As New OleDbCommand
    Dim read As OleDbDataReader

    Private Sub Form1_Load(sender As Object, e As EventArgs) Handles MyBase.Load
        con.Close()
        con.Open()
        cmd = New OleDbCommand("select uname,mark from sub", con)
        read = (cmd.ExecuteReader)
        While (read.Read)
            Chart1.Series("Student").Points.AddXY(read.GetValue(0), read.GetValue(1))

        End While
      

        Chart1.Series("Student").ChartType = System.Windows.Forms.DataVisualization.Charting.SeriesChartType.Pie

        Chart1.Series("Student")("PieLabelStyle") = "Outside"

        Chart1.ChartAreas("ChartArea1").Area3DStyle.Enable3D = True

    End Sub
End Class




5)Save the form then run
